799 bool PumpController::update(const double & _dt, bool & _activate_watering){
800 
801  bool consumed = false;
802  bool fault = false;
803  std::bitset<8> errcode;
804  /*******errcode**********
805  * 00000000
806  * ||||||||->(0) 1 if cmd not consumed
807  * |||||||-->(1) 1 if active, 0 if stopped
808  * ||||||--->(2) 1 if runtime timeout
809  * |||||---->(3) 1 if fault occurred at least once
810  * ||||----->(4)
811  * |||------>(5) 1 if none of avbl pumps was correctly initialized/created
812  * ||------->(6) 1 if controller is in wrong or not avbl mode
813  * |-------->(7) 1 if pumpsCount is 0
814  *************************/
815 
816  if(this->pumpsCount > 0)
817  {
818  switch (this->mode)
819  {
820 
821  case pumpcontrollermode_t::init:
822  errcode.set(6,true);
823  break;
824 
825  case pumpcontrollermode_t::external:
826  if (this->pBinPump != nullptr)
827  {
828  if(_activate_watering)
829  {
830  this->pBinPump->run(_dt, pumpcmd_t::start, consumed);
831  if(consumed == false) errcode.set(0,true);
832  }
833  else
834  {
835  this->pBinPump->run(_dt, pumpcmd_t::stop, consumed);
836  if(consumed == false) errcode.set(0,true);
837  }
838  }
839  else if (this->p8833Pump != nullptr)
840  {
841  if(_activate_watering == true && this->p8833Pump->getState() != pumpstate_t::waiting)
842  {
843  this->p8833Pump->run(_dt, pumpcmd_t::start, consumed, fault);
844  if(consumed == false) errcode.set(0,true);
845  }
846  else if (_activate_watering == true && this->p8833Pump->getState() == pumpstate_t::waiting)
847  {
848  this->p8833Pump->run(_dt, pumpcmd_t::stop, consumed, fault);
849  if(consumed == false) errcode.set(0,true);
850  errcode.set(2,true); //runtime timeout
851  _activate_watering = false;
852  }
853  else if (_activate_watering == false)
854  {
855  this->p8833Pump->run(_dt, pumpcmd_t::stop, consumed, fault);
856  }
857  }
858  else errcode.set(5,true);
859 
860  break;
861 
862  case pumpcontrollermode_t::manual:
863  errcode.set(6,true);
864  break;
865 
866  case pumpcontrollermode_t::automatic:
867  errcode.set(6,true);
868  break;
869 
870  case pumpcontrollermode_t::sleep:
871  errcode.set(6,true);
872  break;
873 
874  default:
875  errcode.set(6,true);
876  break;
877  }
878 
879  if (this->pBinPump != nullptr)
880  {
881  if (this->pBinPump->getState() == pumpstate_t::running) errcode.set(1,true);
882  }
883  else if (this->p8833Pump != nullptr)
884  {
885  if (this->p8833Pump->getState() == pumpstate_t::running) errcode.set(1,true);
886  }
887 
888  if (fault == true){
889  if (++this->pumpFaultOccurenceCnt > 0) errcode.set(3, true);
890  }
891 
892  }
893  else errcode.set(7,true);
894 
895  this->pumpEncodedStatus = static_cast<uint8_t>(errcode.to_ulong());
896 
897  return consumed;
898 }
